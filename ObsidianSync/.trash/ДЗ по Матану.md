# 12.09

$$
\begin{gather}
\lim_{ n \to \infty } a_{n} = + \infty \Leftrightarrow \forall M>0 \; \exists N(M) \in \mathbb{N}: \; \forall n > N(M) \; a_{n} > M  \\
\lim_{ n \to \infty } a_{n} = - \infty \Leftrightarrow \forall M>0 \; \exists N(M) \in \mathbb{N}: \; \forall n > N(M) \; a_{n} < -M
\end{gather}
$$

$$
\begin{gather}
\lim_{ n \to \infty } \left( \frac{(-1)^{n}}{n} + 1 \right) = 1 \\
a_{1} = 0; a_{2} = \frac{1}{2}; a_{3} = \frac{4}{3} \\
\lim_{ n \to \infty } \left( \frac{(-1)^{n}}{n} + 1 \right) = 1 \Leftrightarrow \forall\varepsilon > 0 \; \exists N(\varepsilon) \in \mathbb{N}: \; \forall n > N(\varepsilon) \\
\left\lvert  \frac{(-1)^{n}}{n} + 1 - 1  \right\rvert < \varepsilon \\
\left\lvert  \frac{(-1)^{n}}{n}  \right\rvert < \varepsilon \\
\frac{1}{n} < \varepsilon \\
n > \frac{1}{\varepsilon} \\
\text{Пусть } \varepsilon = 0,5 \implies N = 3
\end{gather}
$$
