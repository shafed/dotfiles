```C++
\#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s;
    cin >> s;
    int pos = s.find_first_of("AOYIUEaoyiue");
    while (pos + 1)
    {
        s.erase(s.begin() + pos);
        pos = s.find_first_of("AOYIUEaoyiue");
    }
    for (char &c : s)
    {   
        cout << '.' << char(tolower(c));
    }
}
```
