#include <bits/stdc++.h>
using namespace std;

int main()
{   
    string s;
    cin >> s;
    vector <int> v;
    for (size_t i = 0; i < s.size(); i += 2)
    {
        v.push_back(stoi(s.substr(i, 1)));
    }
    sort(v.begin(), v.end());
    for (auto i = v.begin(); i != v.end(); i++)
    {
        if (i != v.end() - 1)
        {
            cout << *i << "+";
        }
        else 
        {
            cout << *i;
        }
    }
    return 0;
}
