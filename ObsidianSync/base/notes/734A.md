```C++
\#include <iostream>
\#include <string>
\#include <algorithm>
using namespace std;

int main()
{
    int n;
    string s;
    cin >> n;
    cin >> s;
    int count_A = count(s.begin(), s.end(), 'A');
    int count_D = count(s.begin(), s.end(), 'D');
    if (count_A > count_D) cout << "Anton";
    else if (count_D > count_A) cout << "Danik";
    else cout << "Friendship";
    return 0;
}
```
