$$
\begin{array}{l}

E(f(\xi)) = \sum f(x) * P(\xi = x) \\
D\xi = E\xi^2 - (E\xi)^2 \\
cov (\xi, \eta) = E(\xi * \eta) - E\xi * E\eta \\
\rho(\xi, \eta) = \frac{cov(\xi, \eta)}{\sqrt{ D\xi * D\eta* }} \\

\end{array}
$$

# Неравенство Маркова

$$
\begin{align}
P(\xi \geq c) \leq \frac{E\xi}{c}
\end{align}
$$

# Неравенство Чебышева

$$
\begin{align}
P(|\xi - E\xi| \geq a) \leq \frac{D\xi}{a^2}
\end{align}
$$
# Непрерывные случайные величины


$$
\begin{align*}
&\int_{B} \rho(x)dx = P(\xi \in B) 
\equiv \int_{a}^{b} \rho(x) dx= F(b) - F(a), \space B = [a;b]\\
&Ef(\xi) = \int_{\mathbb{R}} f(x) \rho(x) dx
\end{align*}
$$

# Равномерные случайные величины
$$
\begin{align}
\rho(x) = \frac{1}{b-a}, \space a\leq x\leq b
\end{align}
$$

# Нормальное распределение

$$
\begin{array}{l}

N(a, \sigma^2), [a \in \mathbb{R}, \sigma^2 \geq 0 ] \\
\rho(x) = \frac{1}{\sqrt{ 2 \pi \sigma^2 }} e^{ \frac{-(x-a)^2}{2 \sigma^2} } \\
E\xi = a \\
D\xi = \sigma^2

\end{array}
$$

# Показательное распределение


$$
\begin{array}{l}

\rho(x) = \lambda e^{ -\lambda x }, \space x > 0 \\
E\xi = \frac{1}{\lambda} \\
D\xi = \frac{1}{\lambda^2}

\end{array}
$$

# Пуассоновское распределение

$$
 \begin{array}{l}
 
P(\xi = k) = \frac{\lambda^k}{k!}e^{ -\lambda }, \space k = 0, 1, 2, \dots \\
E\xi = \lambda \\
D\xi = \lambda

\end{array}
$$

# Математическая статистика

## Среднее отклонение

$$
\begin{array}{l}
\overline{x} = \frac{x_{1}+x_{2}+\dots+x_{n}}{n}

\end{array}
$$

## Стандартное отклонение

$$
\begin{align}
\sigma^2 = \frac{(\overline{x} - x_{1})^2 + (\overline{x} - x_{2})^2 + \dots(\overline{x} - x_{n})^2}{n-1}
\end{align}
$$
