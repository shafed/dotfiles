```dataviewjs
// имя текущего файла, например «Русский Номер 07»
const fileName = dv.current().file.name;

// последние две цифры номера
const numStr = fileName.slice(-2);
const num    = Number(numStr);

// вычисляем предыдущий и следующий номера (циклически 01 ↔ 27)
const prev = ((num + 25) % 27 + 1).toString().padStart(2, "0");
const next = ((num % 27)  + 1).toString().padStart(2, "0");

// функция-шаблон для meta-bind-button
const btn = (target) => `
\`\`\`meta-bind-button
label:  "Русский Номер ${target}"
style: primary
class: obsidian
actions:
  - type: open
    link: "[[Русский Номер ${target}]]"
\`\`\``;

// выводим кнопки
dv.paragraph(btn(prev) + '\n\n' + btn(next));
```

# Самые популярные случаи

## Нужно заменить

| Ошибочное                          | Верное                                                    |
| ---------------------------------- | --------------------------------------------------------- |
| имеет роль                         | играет роль                                               |
| сыграло значение                   | имело значение, сыграло роль                              |
| развязался спор                    | завязался спор                                            |
| уделить значение                   | уделить внимание, придать или придавать значение          |
| **создал реакцию**                 | **вызвал реакцию**                                        |
| делал впечатление                  | производил или создавал впечатление                       |
| совершить победу                   | одержать победу, совершить усилие или усилия              |
| одержать первенство                | одержать победу, выиграть первенство                      |
| неослабную помощь                  | посильную или постоянную или необходиму помощь            |
| достигнутыми недостатками          | имеющимися или существующими или выявленными недостатками |
| скоропалительный рост              | стремительный или быстрый рост                            |
| белого листа                       | чистого листа                                             |
| закадычный враг                    | заклятый или злейший враг, закадычный друг                |
| б*о*льшая половина                 | б*о*льшая часть                                           |
| принести положительные последствия | принести положительные результаты                         |
| дешёвые цены                       | низкие цены                                               |
| глубокой юности                    | ранней юности                                             |
| насупить лицо                      | насупить брови или лоб                                    |
| большую половину                   | большую часть                                             |
