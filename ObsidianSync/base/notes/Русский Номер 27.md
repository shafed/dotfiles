```dataviewjs
// имя текущего файла, например «Русский Номер 07»
const fileName = dv.current().file.name;

// последние две цифры номера
const numStr = fileName.slice(-2);
const num    = Number(numStr);

// вычисляем предыдущий и следующий номера (циклически 01 ↔ 27)
const prev = ((num + 25) % 27 + 1).toString().padStart(2, "0");
const next = ((num % 27)  + 1).toString().padStart(2, "0");

// функция-шаблон для meta-bind-button
const btn = (target) => `
\`\`\`meta-bind-button
label:  "Русский Номер ${target}"
style: primary
class: obsidian
actions:
  - type: open
    link: "[[Русский Номер ${target}]]"
\`\`\``;

// выводим кнопки
dv.paragraph(btn(prev) + '\n\n' + btn(next));
```
