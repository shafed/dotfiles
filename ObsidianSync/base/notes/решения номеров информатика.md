1. нужно сбрасывать счетчик и брать левую границу без одного символа
2. в while всегда берем l по левую границу
```python
s = open('24_12476.txt').readline()

l = m = cnt = 0
for r in range(1, len(s)):
    if s[r-1:r+2] in ('ORO', 'ROR'):
        l = r
        cnt = 0
    if s[r-1:r+1] == 'RO': cnt += 1
    while cnt > 21:
        if s[l:l+2] == 'RO': cnt -=1
        l += 1
    if cnt == 21: m = max(m, r - l + 1)
print(m)
```

```python
f = open('26.txt')
n = int(f.readline())
a = [[int(x) for x in s.split()] for s in f]
a.sort(key = lambda x: x[1])

ans = [( i[2], i[1]-i[0]+1 ) for i in a]

for i in range(1, n):
    sti, endi, pri = a[i]
    for j in range(i):
        stj, endj, prj = a[j]
        if endj + 1 <= sti:
            new_time = ans[j][1] + endi - sti + 1
            ans[i] = max( (ans[i][0], ans[i][1]), (ans[j][0] + pri, new_time), key = lambda x: (x[0], -x[1])  )

print( *max( ans, key = lambda x: (x[0], -x[1]) ) )
```
